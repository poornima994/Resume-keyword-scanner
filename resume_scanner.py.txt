# Step 1: Get job description keywords
job_desc = input("Paste the job description here: ").lower()

keywords = ["python", "java", "sql", "cloud", "aws", "cryptography", "git"] 

print("Job keywords we’ll check:", keywords)

# Step 2: Get your resume text
print("\nNow paste your resume text:")
resume_text = input().lower()

# Step 3: Check which keywords are in your resume
found_keywords = []
missing_keywords = []

for word in keywords:
    if word in resume_text:
        found_keywords.append(word)
    else:
        missing_keywords.append(word)

print("\n--- Results ---")
print("Found keywords:", found_keywords)
print("Missing keywords:", missing_keywords)
print(f"Score: {len(found_keywords)}/{len(keywords)}")
